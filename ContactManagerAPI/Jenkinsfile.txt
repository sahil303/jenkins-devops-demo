pipeline {
    agent any
    environment {
        STAGING_SERVER = 'staging-server'
        PRODUCTION_SERVER = 'production-server'
    }

    stages {
        stage('Checkout') {
        agent { label 'build-node' }
            steps {
                git url: 'https://github.com/sahil303/jenkins-devops-demo.git', branch: 'main'
            }
        }

        stage('Restore') {
        agent { label 'build-node' }
            steps {
                echo 'Restoring .NET dependencies...'
                bat 'dotnet restore ContactManagerAPI\ContactManagerAPI.sln'
            }
        }

        stage('Build') {
            agent { label 'build-node' }
            steps {
                bat 'dotnet build --configuration Release ContactManagerAPI\ContactManagerAPI.sln'
            }
        }

        stage('Test') {
            agent { label 'test-node' }
            steps {
                bat 'dotnet test --no-build ContactManagerAPI\ContactManagerAPI.sln'
            }
        }

        stage('Deploy to Staging') {
            steps {
                echo "Deploying to staging environment: ${env.STAGING_SERVER}"
                // Simulate deployment or run a custom script
                bat 'echo Simulating Staging Deployment'
            }
        }
    }

    post {
        failure {
            echo 'Pipeline failed. Triggering notification/rollback if configured.'
        }
    }
}
