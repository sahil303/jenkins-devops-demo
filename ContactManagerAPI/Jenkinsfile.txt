pipeline {
    agent any
    environment {
        STAGING_SERVER = 'staging-server'
        PRODUCTION_SERVER = 'production-server'
    }

    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/sahil303/jenkins-devops-demo.git', branch: 'main'
            }
        }

        stage('Restore') {
            steps {
                echo 'Restoring .NET dependencies...'
                bat 'dotnet restore'
            }
        }

        stage('Build') {
            steps {
                echo 'Building the application...'
                bat 'dotnet build --configuration Release'
            }
        }

        stage('Test') {
            steps {
                echo 'Running unit tests...'
                bat 'dotnet test --no-build --verbosity normal'
            }
        }

        stage('Deploy to Staging') {
            steps {
                echo "Deploying to staging environment: ${env.STAGING_SERVER}"
                // Simulate deployment or run a custom script
                bat 'echo Simulating Staging Deployment'
            }
        }

        stage('Manual Approval') {
            steps {
                input message: 'Approve Deployment to Production?'
            }
        }

        stage('Deploy to Production') {
            steps {
                echo "Deploying to production environment: ${env.PRODUCTION_SERVER}"
                // Simulate production deployment
                bat 'echo Simulating Production Deployment'
            }
        }
    }

    post {
        failure {
            echo 'Pipeline failed. Triggering notification/rollback if configured.'
        }
    }
}
